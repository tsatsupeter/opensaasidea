---
title: "List Comments"
api: "GET /ideas/{id}/comments"
description: "Get all comments for a specific idea."
---

## Path Parameters

<ParamField path="id" type="string" required>
  The idea slug or UUID.
</ParamField>

## Query Parameters

<ParamField query="limit" type="integer" default="50">
  Number of comments to return (max 100).
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of comments to skip.
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of comment objects.
  <Expandable>
    <ResponseField name="id" type="string" />
    <ResponseField name="idea_id" type="string" />
    <ResponseField name="user_id" type="string" />
    <ResponseField name="parent_id" type="string">
      Parent comment ID for replies. `null` for top-level comments.
    </ResponseField>
    <ResponseField name="content" type="string" />
    <ResponseField name="upvotes" type="integer" />
    <ResponseField name="downvotes" type="integer" />
    <ResponseField name="is_edited" type="boolean" />
    <ResponseField name="created_at" type="string" />
    <ResponseField name="updated_at" type="string" />
  </Expandable>
</ResponseField>

## Scope

`comments:read` - $0.01 per call
